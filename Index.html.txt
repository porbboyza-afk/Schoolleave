                                            <div className="flex-1"><h4 className="font-black text-2xl tracking-tight mb-1">{t?.name} ({t?.pos})</h4><p className="mt-6 text-slate-600 bg-slate-50 p-6 rounded-3xl border-l-[10px] border-blue-600 italic font-bold text-lg leading-relaxed shadow-sm">"{req.reason}"</p></div>
                                            <div className="flex gap-4"><button onClick={() => setRequests(requests.map(r => r.id === req.id ? {...r, status: 'rejected'} : r))} className="px-8 py-5 border-2 text-red-600 font-black rounded-2xl text-xs uppercase tracking-widest">Reject</button><button onClick={() => setRequests(requests.map(r => r.id === req.id ? {...r, status: 'approved'} : r))} className="px-14 py-5 bg-emerald-600 text-white font-black rounded-2xl text-xs uppercase shadow-xl hover:bg-emerald-700">Approve</button></div>
                                        </div>
                                    );
                                })}
                                {requests.filter(r => r.status === 'pending').length === 0 && <div className="text-center py-20 text-slate-300 italic font-black uppercase">No Pending Data</div>}
                            </div>
                        )}

                        {isMaster && activeTab === 'manage_all' && (
                            <div className="space-y-6">
                                <div className="bg-amber-600 rounded-[3rem] p-10 text-white shadow-xl flex justify-between items-center relative overflow-hidden"><h3 className="text-3xl font-black italic uppercase relative z-10">ธุรการ: จัดการข้อมูลครูทุกคน</h3><Icon name="shield" size={100} className="absolute right-10 opacity-10" /></div>
                                <div className="bg-white rounded-[2.5rem] border overflow-hidden shadow-sm overflow-x-auto">
                                    <table className="w-full text-left min-w-[800px]"><thead className="bg-slate-50 text-[10px] font-black uppercase tracking-widest"><tr><th className="px-8 py-5">ชื่อผู้ลา</th><th className="px-8 py-5">ประเภท</th><th className="px-8 py-5">ระยะเวลา</th><th className="px-8 py-5">สถานะ</th><th className="px-8 py-5 text-right">ดำเนินการ</th></tr></thead><tbody className="divide-y divide-slate-100 font-bold">
                                        {requests.map(req => {
                                            const t = teachers.find(x => x.id === req.userId);
                                            const isEditing = editingRequest?.id === req.id;
                                            return (
                                                <tr key={req.id} className="hover:bg-slate-50/50">
                                                    <td className="px-8 py-6">{t?.name}</td>
                                                    <td className="px-8 py-6">{isEditing ? <select value={editingRequest.type} onChange={(e) => setEditingRequest({...editingRequest, type: e.target.value})} className="bg-slate-100 p-1 rounded font-bold"><option>ลากิจ</option><option>ลาป่วย</option><option>ลาคลอดบุตร</option><option>ลาช่วยภรรยา</option><option>ลาอุปสมบท</option></select> : req.type}</td>
                                                    <td className="px-8 py-6 text-sm italic">{isEditing ? <div className="flex gap-1"><input type="text" value={editingRequest.startDate} onChange={(e) => setEditingRequest({...editingRequest, startDate: e.target.value})} className="w-24 bg-slate-100 px-1 rounded" /><input type="text" value={editingRequest.endDate} onChange={(e) => setEditingRequest({...editingRequest, endDate: e.target.value})} className="w-24 bg-slate-100 px-1 rounded" /></div> : `${formatDateBE(req.startDate)} - ${formatDateBE(req.endDate)}`}</td>
                                                    <td className="px-8 py-6 uppercase text-[10px]">{isEditing ? <select value={editingRequest.status} onChange={(e) => setEditingRequest({...editingRequest, status: e.target.value})} className="bg-slate-100 p-1 rounded"><option value="pending">PENDING</option><option value="approved">APPROVED</option><option value="rejected">REJECTED</option></select> : req.status}</td>
                                                    <td className="px-8 py-6 text-right">
                                                        {isEditing ? (
                                                            <div className="flex gap-2 justify-end"><button onClick={() => { setRequests(requests.map(r => r.id === editingRequest.id ? editingRequest : r)); setEditingRequest(null); }} className="p-2 bg-emerald-500 text-white rounded-lg"><Icon name="save" size={16} /></button><button onClick={() => setEditingRequest(null)} className="p-2 bg-slate-200 text-slate-500 rounded-lg"><Icon name="x-circle" size={16} /></button></div>
                                                        ) : (
                                                            <div className="flex gap-2 justify-end"><button onClick={() => setEditingRequest({...req})} className="p-2 bg-blue-50 text-blue-600 rounded-lg hover:bg-blue-100"><Icon name="edit-3" size={16} /></button><button onClick={() => setRequestToDelete(req.id)} className="p-2 bg-red-50 text-red-500 rounded-lg hover:bg-red-100"><Icon name="trash-2" size={16} /></button></div>
                                                        )}
                                                    </td>
                                                </tr>
                                            );
                                        })}
                                    </tbody></table>
                                </div>
                            </div>
                        )}
                    </main>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
